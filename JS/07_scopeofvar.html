Scope of variable are listed below with example

  1. Global Scope

    A variable declared at the top of a program or outside of a function is considered a global scope variable.

                let a = "hello";

            function greet () {
                console.log(a);
            }
            
            greet(); // hello

  2. Local Scope

    A variable can also have a local scope, i.e it can only be accessed within a function.

                    let a = "hello";
                function greet() {
                    let b = "World"
                    console.log(a + b);
                }
                greet();
                console.log(a + b); // error

  3. blocked scope variable

    The let keyword is block-scoped (variable can be accessed only in the immediate block).

                    let a = 'Hello';

                function greet() {

                    // local variable
                    let b = 'World';

                    console.log(a + ' ' + b);

                    if (b == 'World') {

                        // block-scoped variable
                        let c = 'hello';

                        console.log(a + ' ' + b + ' ' + c);
                    }

                    // variable c cannot be accessed here
                    console.log(a + ' ' + b + ' ' + c);
                }

                greet();
